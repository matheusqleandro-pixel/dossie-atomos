function entrar() {

  const nome   = document.getElementById('input-nome').value.trim();

  const turma  = document.getElementById('input-turma').value.trim().toUpperCase();

  const grupo  = document.getElementById('input-grupo').value.trim();

  const modelo = document.getElementById('input-modelo').value;

  const err    = document.getElementById('home-error');


  if (!nome || !turma || !grupo || !modelo) {

    err.textContent = "Preencha todos os campos antes de continuar.";

    err.style.display = 'block';

    return;

  }

  err.style.display = 'none';


  state.nome   = nome;

  state.turma  = turma;

  state.grupo  = grupo;

  state.modeloPrincipal = modelo;

  state.modeloIdx = MODELOS.findIndex(m => m.id === modelo);

  if (state.modeloIdx < 0) state.modeloIdx = 0;

  state.secaoIdx = 0;


  const chave = `turma_${turma}/grupo_${grupo}`;

  refGrupo = db.ref(`dossies/${chave}`);


  // Escuta mudanças em tempo real

  if (listener) listener();

  listener = refGrupo.on('value', snap => {

    const data = snap.val() || {};

    const novas = data.respostas || {};

    // Só atualiza campos que não estão focados agora

    Object.entries(novas).forEach(([k, v]) => {

      if (state.answers[k] !== v) {

        state.answers[k] = v;

        const el = document.querySelector(`[data-key="${k}"]`);

        if (el && document.activeElement !== el) el.value = v;

      }

    });

    atualizarUI();

  });


  // Abre tela do app

  mostrarTela('app');

  document.getElementById('h-grupo').textContent = `Turma ${turma} · Grupo ${grupo}`;

  document.getElementById('h-turma-nome').textContent = nome;

  renderModelTabs();

  renderSection();

}


/* ══════════════════════════════════════════════════════════════════

   💾 SALVAR

   ══════════════════════════════════════════════════════════════════ */

function salvar(campo, valor) {

  state.answers[campo] = valor;

  setSyncStatus('saving');

  clearTimeout(saveTimer);

  saveTimer = setTimeout(() => {

    refGrupo.update({ respostas: state.answers })

      .then(() => setSyncStatus('ok'))

      .catch(() => setSyncStatus('error'));

  }, 600);

}


function setSyncStatus(s) {

  const dot  = document.getElementById('sync-dot');

  const text = document.getElementById('sync-text');

  if (s === 'saving') { dot.className = 'sync-dot saving'; text.textContent = 'Salvando...'; }

  else if (s === 'ok') { dot.className = 'sync-dot'; text.textContent = 'Salvo ✓'; }

  else { dot.className = 'sync-dot saving'; text.textContent = 'Erro de conexão'; }

}


function getVal(modeloId, secaoId, campo) {

  return state.answers[`${modeloId}/${secaoId}/${campo}`] || '';

}

function setVal(modeloId, secaoId, campo, valor) {

  salvar(`${modeloId}/${secaoId}/${campo}`, valor);

}

  // Botão prev

  document.getElementById('btn-prev').style.display =

    (state.modeloIdx === 0 && state.secaoIdx === 0) ? 'none' : 'block';


  // Botão next

  const btn = document.getElementById('btn-next');

  const isLast = state.modeloIdx === MODELOS.length - 1 && state.secaoIdx === modelo.secoes.length - 1;

  btn.style.background = isLast ? '#4A235A' : modelo.cor;

}


function renderCampo(modelo, secao) {

  const mid = modelo.id, sid = secao.id;


  if (secao.tipo === 'texto') {

    const val = getVal(mid, sid, 'main');

    return `<textarea rows="${secao.linhas + 1}" data-key="${mid}/${sid}/main"

      oninput="setVal('${mid}','${sid}','main',this.value)"

      placeholder="Escreva sua resposta aqui...">${val}</textarea>`;


}


if (secao.tipo === ‘lista’) {

return secao.itens.map((item, i) => {

const val = getVal(mid, sid, `item_${i}`);

return `<div class="bullet-row"> <span class="bullet-label" style="color:${modelo.cor}">${item}:</span> <input class="answer-input" data-key="${mid}/${sid}/item_${i}" type="text" value="${escHtml(val)}" placeholder="..." oninput="setVal('${mid}','${sid}','item_${i}',this.value)"> </div>`;

}).join(’’);

}


if (secao.tipo === ‘trajetorias’) {

return secao.itens.map((item, i) => {

const vO = getVal(mid, sid, `t${i}_o`);

}).join(’’);

}


if (secao.tipo === ‘camadas’) {

const camadas = [‘K’,‘L’,‘M’,‘N’];

const elementos = [

{ nome: ‘Sódio (Na, Z=11)’,  cams: [‘K’,‘L’,‘M’] },

const v = getVal(mid, sid, `cam_${c}`);

});

html += `</div> <p style="font-size:14px;font-weight:700;color:#555;margin:14px 0 8px">Distribuição eletrônica:</p>`;

html += `<div class="elemento-card"> <div class="elemento-nome" style="color:${modelo.cor}">${el.nome}</div> <div class="elemento-row">`;

function irModelo(idx) {

state.modeloIdx = idx;

state.secaoIdx = 0;

renderModelTabs();

if (idx === 4) renderFinal();

else renderSection();

}


function irSecao(idx) {

state.secaoIdx = idx;

renderSection();

renderModelTabs();

}


function nextSection() {

const modelo = MODELOS[state.modeloIdx];

if (!modelo) { concluir(); return; }

if (state.secaoIdx < modelo.secoes.length - 1) {

state.secaoIdx++;

renderSection();

} else if (state.modeloIdx < MODELOS.length - 1) {

state.modeloIdx++;

state.secaoIdx = 0;

renderModelTabs();

renderSection();

} else {

renderFinal();

renderModelTabs();

}

}


function prevSection() {

if (state.modeloIdx === 4) {

state.modeloIdx = MODELOS.length - 1;

state.secaoIdx = MODELOS[state.modeloIdx].secoes.length - 1;

renderModelTabs();

renderSection();

return;

}

if (state.secaoIdx > 0) {

state.secaoIdx–;

renderSection();

} else if (state.modeloIdx > 0) {

state.modeloIdx–;

state.secaoIdx = MODELOS[state.modeloIdx].secoes.length - 1;

renderModelTabs();

renderSection();

}

}

mostrarTela(‘done’);

}

}


/* ══════════════════════════════════════════════════════════════════

📊 PROGRESSO

══════════════════════════════════════════════════════════════════ */

function temResposta(modeloId, secaoId) {

const prefix = `${modeloId}/${secaoId}/`;

return Object.keys(state.answers).some(k => k.startsWith(prefix) && state.answers[k].trim());

}


function atualizarProgresso() {

let preenchidas = 0;

const total = MODELOS.reduce((s, m) => s + m.secoes.length, 0) + QUADRO_FINAL.linhas.length + 1;

MODELOS.forEach(m => m.secoes.forEach(s => { if (temResposta(m.id, s.id)) preenchidas++; }));

QUADRO_FINAL.linhas.forEach(row => { if (getVal(‘final’, row, ‘resolveu’).trim()) preenchidas++; });

if (getVal(‘final’, ‘reflexao’, ‘main’).trim()) preenchidas++;

document.getElementById(‘progress-fill’).style.width = Math.round(preenchidas / total * 100) + ‘%’;

}


function atualizarUI() {

function corDoModelo(nome) {

const m = MODELOS.find(x => x.nome === nome);

return m ? m.cor : ‘#333’;

}

function emojiDoModelo(nome) {

const m = MODELOS.find(x => x.nome === nome);

return m ? m.emoji : ‘▪️’;

}

function escHtml(str) {

if (!str) return ‘’;

return str.replace(/&/g,’&’).replace(/</g,’<’).replace(/>/g,’>’).replace(/”/g,’"’);

}
